USDT.z - Stablecoin on Binance Smart Chain

üì¶ Smart Contract Architecture

Core Components:

  • Context.sol - Provides msg.sender and msg.data context
  • IERC20.sol - ERC20 interface standard
  • SafeMath.sol - Overflow/underflow protection
  • ERC20.sol - Base ERC20 implementation
  • TeamToken.sol - Main token contract with admin features

Admin Functions:

  • mint(address, uint256) - Mint new tokens (owner only)
  • burn(address, uint256) - Burn tokens (owner only)
  • pauseTrading(bool) - Pause/unpause transfers
  • setTransferFee(uint256) - Set transfer fee (0-10%)
  • setWhitelist(address, bool) - Add/remove whitelist addresses
  • setBlacklist(address, bool) - Add/remove blacklist addresses

// SPDX-License-Identifier: MIT
pragma solidity 0.6.12;

contract TeamToken is ERC20 {
    using SafeMath for uint256;

    address private _owner;
    address public feeWallet;
    uint256 public transferFeePercent;
    bool public paused;

    mapping(address => bool) public isBlacklisted;
    mapping(address => bool) public isWhitelisted;

    event OwnershipTransferred(address indexed previousOwner, address indexed newOwner);
    event FeeWalletUpdated(address indexed newFeeWallet);
    event TransferFeeUpdated(uint256 newFeePercent);

    modifier onlyOwner() {
        require(msg.sender == _owner, "Ownable: caller is not the owner");
        _;
    }

    constructor(
        string memory name_,
        string memory symbol_,
        uint8 decimals_,
        uint256 supply_,
        address initialOwner_,
        address feeWallet_
    ) public ERC20(name_, symbol_, decimals_) {
        require(initialOwner_ != address(0), "Owner cannot be zero");
        require(feeWallet_ != address(0), "Fee wallet cannot be zero");
        
        _owner = initialOwner_;
        feeWallet = feeWallet_;
        transferFeePercent = 0;
        paused = false;

        _mint(initialOwner_, supply_);
    }

    // Admin functions
    function mint(address to, uint256 amount) external onlyOwner { ... }
    function burn(address from, uint256 amount) external onlyOwner { ... }
    function pauseTrading(bool _paused) external onlyOwner { ... }
    function setTransferFee(uint256 feePercent) external onlyOwner { ... }
    function setWhitelist(address account, bool status) external onlyOwner { ... }
    function setBlacklist(address account, bool status) external onlyOwner { ... }
    
    // Full implementation available in repository
}
